37 def load_virtual_fs(self):
38 """
39 @brief Загрузка виртуальной файловой системы из ZIP-архива.
40 """
41 try:
42 with zipfile.ZipFile(self.vfs_path, 'r') as zip_ref:
43 self.vfs = {"": []} # Корневая директория
44 for name in zip_ref.namelist():
45 parts = name.strip("/").split("/")
46 current_level = self.vfs
47
48 for part in parts[:-1]:
49 if part not in current_level:
50 current_level[part] = {"": [], "permissions": "755"} # Подкаталог
51 current_level = current_level[part]
52
53 # Добавляем файл или подкаталог
54 if name.endswith("/"):
55 current_level[parts[-1]] = {"": [], "permissions": "755"} # Пустой каталог
56 else:
57 current_level[""].append({"name": parts[-1], "permissions": "644"}) # Файл
58
59 except FileNotFoundError:
60 print(f"Ошибка: архив {self.vfs_path} не найден.")

106 def execute_command(self, command):
107 """
108 @brief Обрабатывает команды shell.
109 @param command Команда shell.
110 @return Результат выполнения команды.
111 """
112 try:
113 if command.startswith("ls"):
114 result = self.ls()
115 elif command.startswith("cd"):
116 result = self.cd(command.split(" ", 1)[1])
117 elif command.startswith("exit"):
118 result = self.exit()
119 elif command.startswith("chmod"):
120 result = self.chmod(command.split(" ", 2))
121 elif command.startswith("uptime"):
122 result = self.uptime()
123 elif command.startswith("rm"):
124 result = self.rm(command.split(" ", 1)[1])
125 else:
126 result = f"Неизвестная команда: {command}"
127 self.log_action(command, result)
128 except Exception as e:
129 result = f"Ошибка выполнения команды '{command}': {e}"
130 print(result)
131 self.log_action(command, result)
132 return result
