22def get_commits(repo_path: str, since_date: str) -> List[Tuple[str, str, str]]:
23 """
24 @brief Получает список коммитов из репозитория с указанной даты.
25 @param repo_path Путь к репозиторию Git.
26 @param since_date Дата в формате, принимаемом командой 'git log --since', например, '2023-01-01'.
27 @return Список кортежей (хэш коммита, дата, автор).
28 @throws Exception Если команда Git завершилась с ошибкой.
29 """
30 git_command = [
31 "git",
32 "-C",
33 repo_path,
34 "log",
35 "--pretty=format:%H %ct %an",
36 "--since",
37 since_date,
38 ]
39 result = subprocess.run(git_command, stdout=subprocess.PIPE, text=True)
40
41 if result.returncode != 0:
42 raise Exception(f"Error running git command: {result.stderr}")
43
44 commits = result.stdout.splitlines()
45 commit_data = [
46 (
47 c.split()[0],
48 datetime.fromtimestamp(int(c.split()[1]), tz=None).strftime("%Y-%m-%d %H:%M:%S"),
49 c.split()[2]
50 )
51 for c in commits
52 ]
53 return commit_data[::-1] # Reverse to have chronological order

56def build_dependency_graph(commits: List[Tuple[str, str, str]]) -> Digraph:
57 """
58 @brief Создаёт граф зависимостей коммитов с использованием Graphviz.
59 @param commits Список кортежей (хэш коммита, дата, автор) в хронологическом порядке.
60 @return Объект Digraph с построенным графом зависимостей.
61 """
62 dot = Digraph(comment="Git Commit Dependencies")
63
64 for i, (commit, date, author) in enumerate(commits):
65 dot.node(str(i), f"Commit: {commit}\nAuthor: {author}\nDate: {date}")
66 if i > 0:
67 dot.edge(str(i - 1), str(i)) # Connect commits in chronological order
68
69 return dot

83 """

98 commits = get_commits(repo_path, since_date)
